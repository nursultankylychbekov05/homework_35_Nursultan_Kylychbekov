SELECT Название_должности 
FROM Сотрудники, Кадровые_Перемещения, Должности
WHERE Сотрудники.id_сотрудника = Кадровые_Перемещения.Код_сотрудника
  AND Кадровые_Перемещения.Код_должности = Должности.id_должности
  AND Сотрудники.ФИО = 'Сидоров С.С.'
  AND '2026-05-10' >= Кадровые_Перемещения.Дата_начала 
  AND (Кадровые_Перемещения.Дата_конца IS NULL OR '2026-05-10' <= Кадровые_Перемещения.Дата_конца);
